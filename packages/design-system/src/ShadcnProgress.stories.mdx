import { useState, useEffect } from 'react'
import { ShadcnProgress } from './ShadcnProgress'

{/* START README */}
The `Progress` component currently represents a simple re-export of the Shadcn UI Progress component. More information on the structure can be found on the [Shadcn UI Progress documentation](https://ui.shadcn.com/docs/components/progress).
{/* END README */}

export const Default = () => {
  const [progress, setProgress] = useState(13)

useEffect(() => {
const timer = setTimeout(() => setProgress(66), 500)
return () => clearTimeout(timer)
}, [])

return <ShadcnProgress value={progress} className="w-[60%]" />
}
