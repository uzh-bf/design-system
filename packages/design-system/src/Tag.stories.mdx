import Tag from './Tag'

{/* START README */}

The Tag component is a simple pre-styled UI component with pre-defined testing and className props that are consistent with other design-system components.

<br />

The Tag component accepts the following props:

- @param id - The id of the tag.
- @param data - The object of data attributes that can be used for testing (e.g. data-test or data-cy)
- @param label - The label of the tag.
- @param className - The optional className object allows you to override the default styling.

{/* END README */}

export const Default = () => {
  return <Tag label="Default Tag" />
}
