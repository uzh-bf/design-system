import { faQuestionCircle } from '@fortawesome/free-regular-svg-icons'

{/* START README */}
The Navigation component provides an abstracted way to display navigation menu. In addition to the option to display simple buttons (with icons, labels, or both), it also supports dropdowns and nested menus. The component is designed to be flexible and can be used in various contexts, such as in a menubar or as part of a larger navigation system.

<br />

The props for the Navigation component are as follows:

- @param items - The array of items to display in the navigation (required).
- @param className - The optional className object to override default styling for the root.

{/* END README */}

export const Button = () => {
  const items = [{
  type: 'button',
  key: 'navigation-button',
  label: 'Button',
  onClick: () => alert('Button clicked'),
  active: false,
}]

return (

<div>
  <div className="mb-3">
    This is a simple example of a menubar with a single button component. On
    click, the corresponding passed callback function will be executed.
  </div>
  <Navigation items={items} className={{ root: 'bg-slate-100' }} />
</div>
) }
