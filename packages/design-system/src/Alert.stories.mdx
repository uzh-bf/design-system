{/* START README */}
The `Alert` component provides contextual feedback messages to users with clear visual hierarchy and accessibility support. It communicates important information, warnings, errors, or success states in a non-intrusive way.

Use this component when you need to:

- Display status messages after user actions (success, error, warning)
- Show important system notifications or announcements
- Provide contextual help or informational messages
- Alert users to critical issues that require attention

The alert supports different variants (default, destructive) and includes optional icons, titles, and descriptions. It uses semantic HTML and proper ARIA attributes for screen reader accessibility.
{/* END README */}

export const Default = () => {
  return (
    <Alert>
      <Terminal className="h-4 w-4" />
      <AlertTitle>Heads up!</AlertTitle>
      <AlertDescription>
        You can add components to your app using the cli.
      </AlertDescription>
    </Alert>
  )
}

export const Success = () => {
  return (
    <Alert>
      <CheckCircle2Icon />
      <AlertTitle>Success! Your changes have been saved</AlertTitle>
      <AlertDescription>
        This is an alert with icon, title and description.
      </AlertDescription>
    </Alert>
  )
}

export const Icon = () => {
  return (
    <Alert>
      <PopcornIcon />
      <AlertTitle>
        This Alert has a title and an icon. No description.
      </AlertTitle>
    </Alert>
  )
}

export const Destructive = () => {
  return (
    <Alert variant="destructive">

      <AlertTitle>Unable to process your payment.</AlertTitle>

        <p>Please verify your billing information and try again.</p>

          <li>Check your card details</li>
          <li>Ensure sufficient funds</li>
          <li>Verify billing address</li>
        </ul>
      </AlertDescription>
